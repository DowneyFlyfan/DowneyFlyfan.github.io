def upper_bound(u, k, n, v, w):
  if u.weight > k:
    return 0
  else:
    bound = u.value
    wt = u.weight
    j = u.level + 1
    while j < n and wt + w[j] <= k:
      bound += v[j]
      wt += w[j]
      j += 1
    if j < n:
      bound += (k - wt) * (v[j] / w[j])
    return bound

def merge_sort(arr):

def merge(arr, left, mid, right):
    temp = []
    i, j = left, mid + 1
    while i <= mid and j <= right:
        if arr[i] <= arr[j]:
            temp.append(arr[i])
            i += 1
        else:
            temp.append(arr[j])
            j += 1
    while i <= mid:
        temp.append(arr[i])
        i += 1
    while j <= right:
        temp.append(arr[j])
        j += 1
    arr[left:right + 1] = temp

def merge_sort(arr):
    n = len(arr)
    size = 1
    while size < n:
        left = 0
        while left < n:
            mid = left + size - 1
            if mid >= n - 1:
                break
            right = min(mid + size, n - 1)
            merge(arr, left, mid, right)
            left = right + 1
        size *= 2
    return arr
