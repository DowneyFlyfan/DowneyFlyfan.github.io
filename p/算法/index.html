                j += 1

            j += 1
            k += 1
    return arr

def merge(arr, left, mid, right):
    temp = []
    i, j = left, mid + 1
    while i <= mid and j <= right:
        if arr[i] <= arr[j]:
            temp.append(arr[i])
            i += 1
        else:
            temp.append(arr[j])
            j += 1
    while i <= mid:
        temp.append(arr[i])
        i += 1
    while j <= right:
        temp.append(arr[j])
        j += 1
    arr[left:right + 1] = temp

def merge_sort(arr):
    n = len(arr)
    size = 1
    while size < n:
        left = 0
        while left < n:
            mid = left + size - 1
            if mid >= n - 1:
                break
            right = min(mid + size, n - 1)
            merge(arr, left, mid, right)
            left = right + 1
        size *= 2
    return arr
