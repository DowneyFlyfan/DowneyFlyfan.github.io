import random

def knapsack_ga(W, wt, val, n, pop_size=50, elite_size=10, mutation_rate=0.01, generations=100):

'''
W: ËÉåÂåÖÊúÄÂ§ßÊâøÈáç
wt: ÊØè‰∏™Áâ©ÂìÅÁöÑÈáçÈáèÔºåË°®Á§∫‰∏∫‰∏Ä‰∏™ÈïøÂ∫¶‰∏∫nÁöÑÂàóË°®
val: ÊØè‰∏™Áâ©ÂìÅÁöÑ‰ª∑ÂÄºÔºåË°®Á§∫‰∏∫‰∏Ä‰∏™ÈïøÂ∫¶‰∏∫nÁöÑÂàóË°®
n: Áâ©ÂìÅÁöÑÊï∞Èáè
pop_size: ÁßçÁæ§Â§ßÂ∞èÔºåÈªòËÆ§‰∏∫50
elite_size: Á≤æËã±Êï∞ÈáèÔºåÈªòËÆ§‰∏∫10
mutation_rate: ÂèòÂºÇÊ¶ÇÁéáÔºåÈªòËÆ§‰∏∫0.01
generations: ËøõÂåñ‰ª£Êï∞ÔºåÈªòËÆ§‰∏∫100
'''

    # ÂàùÂßãÂåñÁßçÁæ§
    population = []
    for i in range(pop_size):
        individual = [random.randint(0, 1) for _ in range(n)]
        population.append(individual)

    # ÈÄÇÂ∫îÂ∫¶ÂáΩÊï∞
    def fitness(individual):
        return sum(val[i] for i in range(n) if individual[i] == 1)

    # ÈÄâÊã©
    def selection(population):
        return random.sample(population, elite_size) + [max(population, key=fitness) for _ in range(pop_size-elite_size)]

    # ‰∫§Âèâ
    def crossover(ind1, ind2):
        k = random.randint(1, n-1)
        return ind1[:k] + ind2[k:], ind2[:k] + ind1[k:]

    # ÂèòÂºÇ
    def mutation(individual):
        for i in range(n):
            if random.random() < mutation_rate:
                individual[i] = 1 - individual[i]

    # ËøõÂåñ
    for generation in range(generations):
        # ÈÄâÊã©
        selected_population = selection(population)

        # ‰∫§Âèâ
        offspring_population = []
        for i in range(pop_size):
            parent1, parent2 = random.sample(selected_population, 2)
            offspring1, offspring2 = crossover(parent1, parent2)
            offspring_population.append(offspring1)
            offspring_population.append(offspring2)

        # ÂèòÂºÇ
        for individual in offspring_population:
            mutation(individual)

        # Êõ¥Êñ∞ÁßçÁæ§
        population = offspring_population

    # ËøîÂõûÊúÄ‰ºòËß£
    return max(population, key=fitness)

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        # ÊØèÊ¨°ÈÅçÂéÜÂ∞ÜÊúÄÂ§ßÂÄºÁßªÂä®Âà∞Êú´Â∞æ

            else:
                arr[k] = right[j]

def merge(arr, left, mid, right):
    temp = []
    i, j = left, mid + 1
    while i <= mid and j <= right:
        if arr[i] <= arr[j]:
            temp.append(arr[i])
            i += 1
        else:
            temp.append(arr[j])
            j += 1
    while i <= mid:
        temp.append(arr[i])
        i += 1
    while j <= right:
        temp.append(arr[j])
        j += 1
    arr[left:right + 1] = temp

def merge_sort(arr):
    n = len(arr)
    size = 1
    while size < n:
        left = 0
        while left < n:
            mid = left + size - 1
            if mid >= n - 1:
                break
            right = min(mid + size, n - 1)
            merge(arr, left, mid, right)
            left = right + 1
        size *= 2
    return arr
