$$ \begin{gather} \begin{aligned} \frac{ \partial V(x,t) }{ \partial x } = - (R + j \omega L) I(x) \\ \frac{ \partial I(x,t) }{ \partial x } = - (G + j \omega C) V(x) \end{aligned} \end{gather} $$

$$ \begin{gather} \begin{aligned} V_{in} &= V_{in}^+ + V_{in}^- \\ &= V_{in}^+ ( 1 + \Gamma_{in}) \\ &= V_S \frac{ Z_S }{ Z_{in} + Z_S } \\ V_{in}^+ &= \frac{ V_S }{ 2 } \frac{ 1 - \Gamma_S }{ 1 - \Gamma_S \Gamma_{in} } \end{aligned} \end{gather} $$

$$ \begin{gather} \begin{aligned} P_{in} &= \frac{ |V_{in}|^2 }{ 2 Z_0 } \Big[ 1 - | \Gamma_{in} |^2 \Big] \\ &= \frac{ |V_S|^2 }{ 8 Z_0 } \frac{ | 1 - \Gamma_S|^2 }{ |1 - \Gamma_S \Gamma_{in} |^2} ( 1 - | \Gamma_{in}|^2 ) \end{aligned} \end{gather} $$
