def knapsack(weights, values, capacity):
    n = len(weights)
    # dp[i][j]: åœ¨å‰ i ä¸ªç‰©å“ä¸­ï¼ŒèƒŒåŒ…å®¹é‡ä¸º j æ—¶çš„æœ€å¤§ä»·å€¼
    dp = [[0] * (capacity + 1) for _ in range(n + 1)] 

    for i in range(1, n + 1):
        for j in range(1, capacity + 1):
            if weights[i - 1] <= j:
                dp[i][j] = max(dp[i - 1][j], values[i - 1] + dp[i - 1][j - weights[i - 1]])
            else:
                dp[i][j] = dp[i - 1][j]

    selected_items = []
    i, j = n, capacity
    while i > 0 and j > 0:
        if dp[i][j] != dp[i - 1][j]:
            selected_items.append(i)
            j -= weights[i - 1]
        i -= 1

    selected_items.reverse()
    return dp[n][capacity], selected_items

def min_processing_time(products, num_stations):
    if num_stations >= len(products):
        return max(products)
    elif num_stations == 1:
        return sum(products)
   
    n = len(products)
    dp = [[0] * (n + 1) for _ in range(num_stations + 1)]
    for i in range(1, n + 1):
        dp[1][i] = dp[1][i - 1] + products[i - 1]
    for i in range(2, num_stations + 1):
        for j in range(i, n + 1):
            min_time = float("inf")
            for k in range(i - 1, j):
                time = max(dp[i - 1][k], dp[1][j] - dp[1][k])
                min_time = min(min_time, time)
            dp[i][j] = min_time
    return dp[num_stations][n]
