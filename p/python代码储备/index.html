import readline
def input_str():
    user_input = ""
    while True:
        char = input()
        if char == " ":
            user_input = user_input.rstrip(" ")
            break
        elif char == "q":
            user_input = "q"
            break
        else:
            user_input = user_input + char + "\n"
    return user_input
